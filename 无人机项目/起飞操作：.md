#include <stdio.h>

/**

 * 将网格坐标转换为实际显示坐标
 *
 * 坐标转换规则：
 * - 输入坐标 (x, y) 范围: x∈[0,6], y∈[0,8]
 * - 输出坐标 (new_x, new_y) 范围: new_x[25,325], new_y[25,425]
 * - (0,0) -> (325, 425)
 * - (6,8) -> (25, 25)
 * ---
 * 地图 (x,y)
 * ----------------------------------
 * (0,0)                         (0,8)
 *
 *
 * (6,0)                         (6,8)起飞点
 * -----------------------------------
 * @param x 网格x坐标 (行索引)
 * @param y 网格y坐标 (列索引)
 * @param new_x 转换后的给飞控x坐标
 * @param new_y 转换后的给飞控y坐标
 */

void convert_coordinates(int x, int y, int *new_x, int *new_y)

{

    *new_x = 325 - x * 50;

    *new_y = 425 - y * 50;

}
