# 1. 看看哪些目录占空间最大（前10大）
sudo du -h --max-depth=1 / | sort -hr | head -10

# 2. 清理包管理缓存（最常用，能腾出几百M到几G）
sudo apt autoclean
sudo apt autoremove -y
sudo apt clean

# 3. 清理旧的内核（开发板常年积累，轻松几百M）
sudo apt autoremove --purge

# 4. 查找大于100M的大文件
sudo find / -type f -size +100M -exec ls -lh {} \; | awk '{ print $9 ": " $5 }'

# 5. 清日志（如果你不需要历史日志）
sudo journalctl --vacuum-size=50M   # 保留最近50M日志
# 或者直接砍掉旧日志
sudo find /var/log -type f -name "*.log" -exec truncate -s 0 {} \;

# 6. 删除用户缓存
rm -rf ~/.cache/*

# 7. 如果你装了docker/container（很多AI开发板都有）
docker system prune -a   # 慎重，会删掉所有未运行的镜像
