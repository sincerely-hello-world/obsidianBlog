```
ros2 pkg # 总命令
```

创建功能包
```
ros2 pkg create 包名
ros2 pkg create 包名 --build-type ament_python
```


功能包，类似于python的库包， 放置在项目目录src下
```bash
├──Workplace工作区顶层目录
	├── build      # colcon build 构建中间文件（算是缓存、临时）
	├── install    # colcon build 构建后文件（部署时，调用这个）
	├── log        # colcon build 构建日志
	└── src        # 存放源码  
```


```yaml
~/ros2_ws/                     # ← 工作区根目录（通常命名为 ros2_ws、my_ros2_ws 等）
│
├── src/                  # ← 源代码目录：所有 ROS 2 包（packages）都放在这里
│   ├── package_1/             # ← 一个 ROS 2 功能包（例如：my_robot_bringup）
│   │   ├── CMakeLists.txt     # ← C++ 包的构建配置文件（即使纯 Python 包也需要）
│   │   ├── package.xml        # ← 包的元信息：名称、版本、依赖、作者等
│   │   ├── setup.py           # ← Python 包的安装脚本（用于纯 Python 或混合包）
│   │   ├── setup.cfg          # ← Python 包配置（可选，但推荐）
│   │   ├── resource/          # ← （可选）资源文件，如图标（仅当需要时）
│   │   ├── launch/            # ← 启动文件目录（.py 或 .xml/.yaml 格式）
│   │   ├── config/            # ← 配置文件目录（如 YAML 参数文件）
│   │   ├── scripts/           # ← 可执行 Python 脚本（需在 setup.py 中声明）
│   │   ├── nodes/             # ← （常见约定）存放节点源码（非强制）
│   │   └── test/              # ← 单元测试或集成测试代码
│   │
│   └── package_2/             # ← 另一个功能包（例如：sensor_driver）
│       └── ...                # ← 结构同上
│
├── build/                 # ← 构建中间文件目录（由 colcon 自动生成）
│   ├── package_1/             # ← 每个包的编译中间产物（CMake 缓存、目标文件等）
│   └── ...
│
├── install/               # ← 安装目录（由 colcon 自动生成）
│   ├── package_1/             # ← 每个包安装后的文件结构（模拟系统安装路径）
│   │   ├── lib/               # ← 可执行文件（C++ 二进制或 Python 脚本链接）
│   │   ├── share/             # ← 共享资源：launch 文件、config、package.xml 等
│   │   └── local_setup.bash   # ← 用于 source 的环境设置脚本（关键！）
│   ├── setup.bash             # ← 主环境设置脚本（source 此文件激活工作区）
│   └── ...
│
└── log/                   # ← 构建日志目录（colcon build 的输出日志）
    └── latest/                # ← 最近一次构建的日志
```
